import numpy as np
import matplotlib.pyplot as plt
from sklearn.metrics import roc_curve, auc, confusion_matrix, accuracy_score, precision_recall_fscore_support

# Set the random seed for reproducibility
np.random.seed(0)

n = 1000
isCorrect = np.random.choice([0, 1], size=n)
UQ = np.where(isCorrect==1, np.random.beta(2, 9, n), np.random.beta(5, 2, n))

Determine the threshold¶

In [ ]:
fpr, tpr, thresholds = roc_curve(isCorrect, 1-UQ)  # Compute ROC curve with inverted UQ scores
roc_auc = auc(fpr, tpr)  # Compute AUC
optimal_idx = np.argmax(tpr - fpr)  # Find index of optimal threshold
optimal_threshold = 1 - thresholds[optimal_idx]  # Get optimal threshold, and invert it back
print(f'Optimal threshold is {optimal_threshold}')

predicted = np.array([1 if uq < optimal_threshold else 0 for uq in UQ])

# Evaluate the performance
cm = confusion_matrix(isCorrect, predicted)
accuracy = accuracy_score(isCorrect, predicted)
precision, recall, f1, _ = precision_recall_fscore_support(isCorrect, predicted, average='binary')

print(cm)
